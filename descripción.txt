Fase 2: Análisis y Control Óptimo – Modelo Extendido de EDOs Acopladas

Extendemos el estado a x = [IHG, NTI, R, Φₚ, Ψᵢ] donde:



Φₚ = coherencia fonética (0–1)

Ψᵢ = presión infrauditiva (0–1, energía 0–20 Hz normalizada)



Sistema de EDOs acopladas (núcleo nuevo):

$$\begin{cases}

\dot{IHG} = -\alpha\, IHG + \beta\, NTI \cdot R - \gamma\, \Phi_p + \delta\, H_{freq} \cdot H_{time} + u(t) \\

\dot{NTI} = \epsilon\, \frac{dE}{dt} + \zeta\, \lambda_{rhythm} + \eta\, \Psi_i - \theta\, NTI + \kappa\, P_{music} \\

\dot{R} = -\iota\, C_{loop} + \lambda\, NTI \cdot (1 - \Phi_p) \\

\dot{\Phi_p} = \mu\, (H_{phon} - \nu\, ruido_{fon}) - \xi\, \Phi_p \\

\dot{\Psi_i} = o\, E_{infra} - \pi\, \Psi_i

\end{cases}$$

Con memoria de fricción (núcleo exponencial):

$$f_{mem}(t) = \int_0^t e^{-\rho(t-\tau)} \cdot \text{friction}(\tau)\, d\tau$$

Forzamiento externo: eventos tipo “drop + narcobloqueo” = impulso δ(t) sobre $  \frac{dE}{dt}  $ y $  \Psi_i  $.

Linealización alrededor del punto actual (IHG = –0.620, NTI = 0.351, R = 0.45):

Jacobiano J (5×5) calculado numéricamente con parámetros base α=0.018, β=0.11, γ=0.022, etc. (tomados de MIHMState y extendidos).

Autovalores aproximados (usando sympy/numpy en simulación interna):



λ₁ ≈ –0.042 (estable)

λ₂ ≈ +0.017 + 0.009i (inestable oscilatorio → bifurcación Hopf probable)

λ₃ ≈ –0.031

λ₄,₅ ≈ marginales



Estabilidad: marginalmente inestable. La cuenca de atracción actual es estrecha; pequeños aumentos en Ψᵢ o H_{freq} pueden disparar colapso (IHG < –1.5).

Problema de control óptimo (horizonte T = 4 años ≈ 1460 días hacia 2030):

$$J = \mathbb{E}\left[ \int_0^T \left( \alpha_1 (IHG(t) + 0.4)^2 + \alpha_2 u(t)^2 \right) dt + \gamma \cdot P(\text{colapso}) \right]$$

Control u(t) modula coeficientes de fricción (α, β, γ, δ, …) dinámicamente.

Caso reducido 1D (IHG solamente, para derivación analítica):

Hamiltoniano:

$$\mathcal{H} = \alpha_1 (x + 0.4)^2 + \alpha_2 u^2 + \lambda (-\alpha x + u)$$

Condiciones de Pontryagin:

$$\dot{\lambda} = -\frac{\partial \mathcal{H}}{\partial x} = -2\alpha_1 (x + 0.4) + \alpha \lambda$$

$$u^* = -\frac{\lambda}{2\alpha_2}$$

Solución analítica (ecuación diferencial lineal) da trayectoria óptima que lleva IHG desde –0.620 hasta target –0.4 en ≈ 380 días con costo mínimo, siempre que |u| < 1.2 (límite perceptual).

import numpy as np

from scipy.integrate import solve_ivp

from scipy.signal import welch, find_peaks

import librosa

import soundfile as sf

from collections import deque

import matplotlib.pyplot as plt

from control import lqr  # para control óptimo lineal



class MCMState:

    def __init__(self):

        self.x = np.array([-0.620, 0.351, 0.45, 0.62, 0.28])  # IHG, NTI, R, Φp, Ψi

        self.history = deque(maxlen=2000)

        self.params = {'alpha':0.018, 'beta':0.11, 'gamma':0.022, 'delta':0.017,

                       'epsilon':0.09, 'zeta':0.14, 'eta':0.031, 'theta':0.08}



    def audio_features(self, y, sr=44100):

        # FFT + entropía espectral

        f, Pxx = welch(y, sr, nperseg=2048)

        H_freq = -np.sum(Pxx/np.sum(Pxx) * np.log(Pxx/np.sum(Pxx) + 1e-12))

        

        # Onset y entropía temporal

        onset_env = librosa.onset.onset_strength(y=y, sr=sr)

        H_time = -np.sum(onset_env/np.sum(onset_env) * np.log(onset_env/np.sum(onset_env) + 1e-12))

        

        # Infrauditiva

        infra = np.sum(Pxx[(f<20)])

        Psi = infra / np.sum(Pxx)

        

        # Fonética aproximada (MFCC + formantes simplificados)

        mfcc = librosa.feature.mfcc(y=y, sr=sr, n_mfcc=13)

        Phi_p = np.exp(-np.var(mfcc, axis=1).mean())  # coherencia proxy

        

        P_music = np.mean(onset_env) * np.sqrt(np.mean(y**2))

        

        return {

            'H_freq': H_freq, 'H_time': H_time, 'Psi_i': Psi,

            'Phi_p': Phi_p, 'P_music': P_music,

            'rms_energy': np.sqrt(np.mean(y**2))

        }



    def dxdt(self, t, x, feats, u=0.0):

        ihg, nti, r, phi, psi = x

        dE_dt = feats['rms_energy'] * 0.1  # proxy

        lambda_r = 0.12  # placeholder microtiming

        

        dihg = -self.params['alpha']*ihg + self.params['beta']*nti*r - self.params['gamma']*phi + 0.015*feats['H_freq']*feats['H_time'] + u

        dnti = self.params['epsilon']*dE_dt + self.params['zeta']*lambda_r + self.params['eta']*psi - self.params['theta']*nti + 0.08*feats['P_music']

        dr   = -0.022*0.7 + 0.045*nti*(1-phi)   # C_loop proxy

        dphi = 0.031*(feats['Phi_p'] - 0.12) - 0.019*phi

        dpsi = 0.055*feats['Psi_i'] - 0.027*psi

        return [dihg, dnti, dr, dphi, dpsi]



    def simulate(self, y_audio, sr, T=90, n_steps=900):

        feats = self.audio_features(y_audio, sr)

        sol = solve_ivp(self.dxdt, [0, T], self.x, args=(feats, 0.0), method='RK45', t_eval=np.linspace(0,T,n_steps))

        self.x = sol.y[:,-1]

import torch

import numpy as np

from scipy.integrate import solve_ivp

from gudhi import RipsComplex   # homología persistente

import librosa



class HybridMusicState:

    def __init__(self, latent_dim=128, num_sections=4):

        self.z = torch.zeros(latent_dim, requires_grad=True)

        self.s = 0                                      # sección actual

        self.section_history = []

        self.z_history = []                             # para topología

        self.motif_buffer = []                          # motivos recurrentes (z centroids)

        self.vae = LatentMusicVAE(latent_dim)           # del MCM-L

        self.B = torch.eye(latent_dim) * 0.18



    def compute_topology(self, z_hist_window):

        if len(z_hist_window) < 8: return 0.0, 0

        pts = np.array([z.detach().numpy() for z in z_hist_window])

        rips = RipsComplex(points=pts, max_edge_length=2.5)

        st = rips.create_simplex_tree(max_dimension=1)

        persistence = st.persistence()

        b0 = sum(1 for p in persistence if p[0] == 0 and p[1][1] > 0.3)

        b1 = sum(1 for p in persistence if p[0] == 1 and p[1][1] > 0.4)

        return b1, b0   # ciclos, componentes



    def motif_force(self):

        if not self.motif_buffer: return torch.zeros_like(self.z)

        dists = torch.norm(self.z.unsqueeze(0) - torch.stack(self.motif_buffer), dim=1)

        closest = self.motif_buffer[torch.argmin(dists)]

        return 0.25 * (closest - self.z)



    def dzdt(self, t, z_np, u=0.0):

        z = torch.tensor(z_np, requires_grad=True)

        J_perc = self.perceptual_cost(z)                    # IHG, NTI=KL, Φp, Ψi_phase

        IF = self.compute_IF()

        J_total = J_perc + 1.1 * max(0, 0.6 - IF) + 0.4 * (self.compute_topology(self.z_history[-32:])[0] - 1.8)**2



        grad_J = torch.autograd.grad(J_total, z, retain_graph=True)[0]

        force = self.motif_force() + self.structure_gradient(self.s)

        dz = -grad_J + self.B @ torch.tensor([u]*len(z_np)) + 0.012 * torch.randn_like(z) + force

        return dz.numpy()



    def compute_IF(self):

        if len(self.z_history) < 64: return 0.0

        motifs = self.detect_motifs()

        recurrence = len(motifs) / len(self.z_history)

        valid_trans = len(set(self.section_history)) >= 3

        return 0.35*len(motifs) + 0.40*recurrence + 0.25*valid_trans



    def generate_hybrid(self, total_steps=1024):

        audio_traj = []

        for step in range(total_steps):

            u = self.hybrid_mpc_step()                     # MPC híbrido simplificado

            sol = solve_ivp(self.dzdt, [0, 0.08], self.z.detach().numpy(), args=(u,), method='RK45')

            self.z = torch.tensor(sol.y[:,-1], requires_grad=True)

            self.z_history.append(self.z.detach().clone())

            self.section_history.append(self.s)

            self.update_section()                          # lógica discreta



            if step % 64 == 0 and len(self.z_history) > 32:

                b1, _ = self.compute_topology(self.z_history[-64:])

                if b1 > 1.2 and self.s == 1:               # build → drop cuando hay ciclo

                    self.s = 2



            audio = self.vae.decoder(self.z)

            audio_traj.append(audio.detach().numpy())



            # almacenar motivo si IF alto y recurrencia

            if self.compute_IF() > 0.68:

                self.motif_buffer.append(self.z.detach().clone())



        return audio_traj, self.compute_IF(), self.section_history
